CREATE PROCEDURE GETALLDATACUSTOMER
AS
BEGIN

select  Custd.CustId as [Customer ID] ,CustName AS Name ,CustCompName AS [Compnay Name] ,CustAddress AS Address,CustCity AS City, CustState AS State ,CustZip AS Zip ,CustCountry AS Country ,CustEmail AS [E-Mail Address] , CustWebAddress AS [Web Address],CustPhone AS Phone ,CustMobile AS Mobile ,CustFax AS Fax ,CustPanNo AS [PAN NO],CustGstNo AS [GST NO],Custad.CustOpeningBalance AS [Opening Balance] , Custad.CustCurrentBalance AS [Current Ballance],CustDesc AS Description from  CustomerDetails Custd join    CustomerAccountDetails  Custad on Custd.CustID=Custad.CustID

END
go

create procedure GETALLDATAVENDER
